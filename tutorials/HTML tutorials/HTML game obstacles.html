Menambahkan Beberapa Rintangan

Sekarang kita ingin menambahkan beberapa rintangan untuk game kita.

Tambahkan komponen baru untuk area permainan. Buat itu menjadi green, 10px wide, 200px high, dan posisinya di 300px ke kanan dan 120px ke bawah.

Perbaharui juga komponen obstacle pada setisp waktu:

Contoh:

var myGamePiece;
var myObstacle;

function startGame() {
myGamePiece = new component(30, 30, "red", 10, 120);
myObstacle = new component(10, 200, "green", 300, 120);
myGameArea.start();
}

function updateGameArea() {
myGameArea.clear();

Hit The Obstacle = Game Over

Pada contoh diatas, Tidak ada yang terjadu ketika kamu menambrak obstacle. Pada sebuah game, itu sangat tidak memuaskan.

Bagaimana kita tahu jika red square kita menabrak obstacle?

Buatlah metode baru dalam component constructor, bahwa mengetes jika komponen bertabrakan dengan komponen lainnya. Metode ini harus dipanggil setiap kali frame diperbaharui, 50 time per second.

Tambahkan juga metode stop() ke myGameArea object, yang mana membersihkan interval 20 milidetik.

Contoh:

var myGameArea = {
canvas : document.createElement("canvas"),
start : function() {
this.canvas.width = 480;
this.canvas.height = 270;
this.context = this.canvas.getContext("2d");
document.body.insertBefore(this.canvas, document.body.childNodes[0]);
this.interval = setInterval(updateGameArea, 20);
},
clear : function() {
this.context.clearRect(0, 0, this.canvas.width, this.canvas.height);
},
stop : function() {
clearInterval(this.interval);
}
}

function component(width, height, color, x, y) {
this.width = width;
this.height = height;
this.speedX = 0;
this.speedY = 0;
this.x = x;
this.y = y;
this.update = function() {
ctx = myGameArea.context;
ctx.fillStyle = color;
ctx.fillRect(this.x, this.y, this.width, this.height);
}
this.newPos = function() {
this.x += this.speedX;
this.y += this.speedY;
}
this.crashWith = function(otherobj) {
var myleft = this.x;
var myright = this.x + (this.width);
var mytop = this.y;
var mybottom = this.y + (this.height);
var otherleft = otherobj.x;
var otherright = otherobj.x + (otherobj.width);
var othertop = otherobj.y;
var otherbottom = otherobj.y + (otherobj.height);
var crash = true;
if ((mybottom < othertop) ||
(mytop > otherbottom) ||
(myright < otherleft) ||
(myleft > otherright)) {
crash = false;
}
return crash;
}
}

function updateGameArea() {
if (myGamePiece.crashWith(myObstacle)) {
myGameArea.stop();
} else {
myGameArea.clear();
myObstacle.update();
myGamePiece.newPos();
myGamePiece.update();
}
}

Moving Obstacle

Pergerakan Rintangan

Obstacles (rintangan) tidak berbahaya ketika ia statis (tetap/diam), jadi kita kita ingin rintangan menjadi bergerak.

Ubahlah nilai properti dari myObstacle.x di setiap perbaharuan:

Contoh:

function updateGameArea() {
if (myGamePiece.crashWith(myObstacle)) {
myGameArea.stop();
} else {
myGameArea.clear();
myObstacle.x += -1;
myObstacle.update();
myGamePiece.newPos();
myGamePiece.update();
}
}

Multiple Obstacles

Rintangan Lebih dari satu

Bagaiaman tentang penambahan multiple obstacles?

Untuk itu kita membutuhkan sebuah properti untuk penghitungan frames, dan sebuah metode untuk mengeksekusi sesuatu saat sebuah nilai/rate frame yang diberikan.

Contoh:

var myGameArea = {
canvas : document.createElement("canvas"),
start : function() {
this.canvas.width = 480;
this.canvas.height = 270;
this.context = this.canvas.getContext("2d");
document.body.insertBefore(this.canvas, document.body.childNodes[0]);
this.frameNo = 0;
this.interval = setInterval(updateGameArea, 20);
},
clear : function() {
this.context.clearRect(0, 0, this.canvas.width, this.canvas.height);
},
stop : function() {
clearInterval(this.interval);
}
}

function everyinterval(n) {
if ((myGameArea.frameNo / n) % 1 == 0) {return true;}
return false;
}

Fungsi everyinterval mengembalikan true jika kesamaan posisi framenumber saat ini dengan interval yang diberikan.

Untuk mendefinisikan multiple obstacle, pertama-tama deklarasikan variabel obstacle sebagai sebuah array.

Kedua, kita perlu membuat beberapa pengubahan dalam fungsi updateGameArea

Contoh:

var myGamePiece;
var myObstacles = [];

function updateGameArea() {
var x, y;
for (i = 0; i < myObstacles.length; i += 1) {
if (myGamePiece.crashWith(myObstacles[i])) {
myGameArea.stop();
return;
}
}
myGameArea.clear();
myGameArea.frameNo += 1;
if (myGameArea.frameNo == 1 || everyinterval(150)) {
x = myGameArea.canvas.width;
y = myGameArea.canvas.height - 200
myObstacles.push(new component(10, 200, "green", x, y));
}
for (i = 0; i < myObstacles.length; i += 1) {
myObstacles[i].x += -1;
myObstacles[i].update();
}
myGamePiece.newPos();
myGamePiece.update();
}

Pada fungsi updateGameArea kita harus me-loop (menaklukan/menyimpulkan) melewati setiap obstacle untuk melihat jika ada suatu tabrakan. Jika ada sebuah tabrakan (crash), fungsi updateGameArea akan berhenti dan tidak ada lagi penggambaran yang dilakukan.

Fungsi updateGameArea menghitung frame dan menambahkan sebuah obstacle untuk setiap frame ke 150 (150th frame).

Obstacles of Random Size

Rintangan tentang Ukuran Acak

Untuk membuat game sebuah bit lebih sulit dan menyenangkan, kita akan mengirim dalam obstacles dari ukuran acak, jadi bahwa red square harus bergerak keatas atau ke bawah agar tidak menambrak.

Contoh:

function updateGameArea() {
var x, height, gap, minHeight, maxHeight, minGap, maxGap;
for (i = 0; i < myObstacles.length; i += 1) {
if (myGamePiece.crashWith(myObstacles[i])) {
myGameArea.stop();
return;
}
}
myGameArea.clear();
myGameArea.frameNo += 1;
if (myGameArea.frameNo == 1 || everyinterval(150)) {
x = myGameArea.canvas.width;
minHeight = 20;
maxHeight = 200;
height = Math.floor(Math.random()*(maxHeight-minHeight+1)+minHeight);
minGap = 50;
maxGap = 200;
gap = Math.floor(Math.random()*(maxGap-minGap+1)+minGap);
myObstacles.push(new component(10, height, "green", x, 0));
myObstacles.push(new component(10, x - height - gap, "green", x, height + gap));
}
for (i = 0; i < myObstacles.length; i += 1) {
myObstacles[i].x += -1;
myObstacles[i].update();
}
myGamePiece.newPos();
myGamePiece.update();
}
